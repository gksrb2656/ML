def gender(data):
    gender_count = data['gender'].value_counts()
    gender_count.plot.pie(y='wine_group', autopct = "%.1f%%")
    
def age(data):
    age_count = data['age'].value_counts()
    age_count.plot.pie(y='wine_group', autopct = "%.1f%%")
    
def job(data):
    job_count = data['job'].value_counts()
    job_count.plot.pie(y='wine_group', autopct = "%.1f%%")

    marital_count = data['marital'].value_counts()
